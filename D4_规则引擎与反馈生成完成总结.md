### 1. 三维度评分模型

#### 权重配置
```python
DIMENSION_WEIGHTS = {

    "pose_standardization": Decimal("0.4"),  # 姿态规范性 40%

```

#### 综合评分公式
```python
total_score = (
    completeness_score * 0.4 +
    standardization_score * 0.4 +
    timeliness_score * 0.2
)
```

### 2. 动作完整性计算

#### 步骤权重（灭火器操作）
```python
ACTION_STEP_WEIGHTS = {
    "准备阶段": Decimal("0.15"),
    "提灭火器": Decimal("0.20"),
    "拔保险销": Decimal("0.25"),  # 最关键步骤
    "握喷管": Decimal("0.15"),
    "瞄准火源": Decimal("0.15"),
    "压把手": Decimal("0.10"),
}
```

#### 计算方法
```python
def _calculate_action_completeness(self, action_scores):
    weighted_sum = 0
    for step_name, score in action_scores:
        weight = ACTION_STEP_WEIGHTS[step_name]
        weighted_sum += score * weight
    return weighted_sum
```

## API 使用示例

### 基本使用

```python
from app.ai.rule_engine import RuleEngine
from app.ai.feedback_generator import FeedbackGenerator

# 初始化
rule_engine = RuleEngine()
feedback_generator = FeedbackGenerator()

# 准备数据
action_scores = {
    "step_scores": {
        "step1": {"step_name": "准备阶段", "score": 85},
        "step2": {"step_name": "提灭火器", "score": 90},
    }
}

pose_scores = {
    "step_scores": {
        "step1": {"step_name": "准备阶段", "score": 88, "weight": 0.2},
        "step2": {"step_name": "提灭火器", "score": 92, "weight": 0.15},
    },
    "frame_count": 30
}

duration = Decimal("125.5")

# 1. 规则引擎评估
evaluation = await rule_engine.evaluate(
    action_scores=action_scores,
    pose_scores=pose_scores,
    duration_seconds=duration,
    training_type="fire_extinguisher"
)

print(f"总分：{evaluation['total_score']}")
print(f"等级：{evaluation['performance_level']}")

# 2. 生成反馈
feedback = feedback_generator.generate_feedback(
    evaluation_result=evaluation,
    action_logs=None,
    pose_details=pose_scores
)

print(f"总体反馈：{feedback['overall_feedback']}")
print(f"改进建议：{feedback['suggestions']}")
```
